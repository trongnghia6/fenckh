CREATE TABLE quychuan (
    Khoa VARCHAR(255),				-- Khoa
    Dot VARCHAR(255),				-- Đợt 
    KiHoc INT,						-- Kì
    NamHoc INT,						-- Năm
    GiaoVien VARCHAR(255),                     -- Tên giáo viên
    GiaoVienGiangDay VARCHAR(255),	       -- Tên giáo viên giảng môn này
    MoiGiang BOOLEAN DEFAULT FALSE,            -- Mời giảng (TRUE/FALSE, mặc định FALSE)
    SoTinChi INT,                              -- Số tín chỉ (Số TC)
    MaHocPhan VARCHAR(255),                    -- Mã Học Phần
    LopHocPhan VARCHAR(255),                   -- Tên lớp học phần
    TenLop  VARCHAR(255),						-- Tên lớp
    BoMon VARCHAR(255),							-- Bộ Môn
    LL INT,                                    -- LL (để theo yêu cầu của bạn)
    SoTietCTDT INT,                            -- Số tiết chương trình đào tạo (CTĐT)
    HeSoT7CN DECIMAL(3,2),                     -- Hệ số thứ 7/chủ nhật (T7/CN)
    SoSinhVien INT,                            -- Số lượng sinh viên (Số SV)
    HeSoLopDong DECIMAL(3,2),                  -- Hệ số lớp đông
    QuyChuan INT,                              -- Quy chuẩn
    GhiChu VARCHAR(255),		         -- Ghi chú
    KhoaDuyet BOOLEAN DEFAULT FALSE,           -- Mặc định FALSE
    DaoTaoDuyet BOOLEAN DEFAULT FALSE,         -- Mặc định FALSE
    TaiChinhDuyet BOOLEAN DEFAULT FALSE,       -- Mặc định FALSE
    NgayBatDau DATE,                           -- Ngày bắt đầu
    NgayKetThuc DATE,                          -- Ngày kết thúc
    ID INT AUTO_INCREMENT NOT NULL UNIQUE      -- ID
);



CREATE TABLE tam (
    Khoa VARCHAR(255),				-- Khoa
    Dot VARCHAR(255),					-- Đợt
    Ki VARCHAR(255),					-- kì
    Nam VARCHAR(255),					-- Năm	
    GiaoVien VARCHAR(255),                     -- Tên giáo viên
    SoTinChi INT,                              -- Số tín chỉ (Số TC)
    LopHocPhan VARCHAR(255),                   -- Tên lớp học phần
    LL INT,                                    -- LL (để theo yêu cầu của bạn)
    SoTietCTDT INT,                            -- Số tiết chương trình đào tạo (CTĐT)
    HeSoT7CN DECIMAL(3,2),                     -- Hệ số thứ 7/chủ nhật (T7/CN)
    SoSinhVien INT,                            -- Số lượng sinh viên (Số SV)
    HeSoLopDong DECIMAL(3,2),                  -- Hệ số lớp đông
    QuyChuan INT,                              -- Quy chuẩn
    GhiChu VARCHAR(255)		         -- Ghi chú
);

UPDATE quychuan
SET MaHocPhan = CONCAT(Khoa, id);
